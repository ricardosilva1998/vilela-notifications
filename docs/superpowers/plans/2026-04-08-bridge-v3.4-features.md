### Task 2: Wind Arrow Smoothing

**Files:**
- Modify: `bridge/overlays/wind.html` (lines 183-198 lerp animation)

- [ ] **Step 1: Replace fixed lerp factor with delta-time-based interpolation**

Replace the current `animateArrow` function:

```js
let lastFrameTime = 0;
function animateArrow(timestamp) {
  if (!lastFrameTime) lastFrameTime = timestamp;
  const dt = Math.min((timestamp - lastFrameTime) / 1000, 0.1); // cap at 100ms
  lastFrameTime = timestamp;

  let diff = targetAngle - currentAngle;
  if (diff > 180) diff -= 360;
  if (diff < -180) diff += 360;
  // Smooth factor: ~0.12 at 60fps (16.6ms) scales with actual frame time
  const factor = 1 - Math.pow(0.001, dt);
  currentAngle += diff * factor;
  currentAngle = ((currentAngle % 360) + 360) % 360;
  if (arrowGroup) arrowGroup.setAttribute('transform', `rotate(${currentAngle}, 60, 60)`);
  requestAnimationFrame(animateArrow);
}
requestAnimationFrame(animateArrow);
```

This uses exponential decay (`Math.pow(0.001, dt)`) which is frame-rate independent — same smoothness at 30fps and 144fps.

onData('standings', data => {
  const focused = data.find(d => d.isSpectated) || data.find(d => d.isPlayer);
  if (!focused) return;
  // Get cars in same class sorted by position
  const classmates = data.filter(d => d.carClass === focused.carClass)
    .sort((a, b) => a.classPosition - b.classPosition);
  // Find cars behind the focused driver
  const behindCars = classmates.filter(d => d.classPosition > focused.classPosition);
  // How many would pass during a pit stop?
  const pitDelta = _pitDelta; // from session data
  const wouldPass = behindCars.filter(d => {
    // Gap from focused to this car (time gap)
    const gap = d.gapToLeader - focused.gapToLeader;
    return gap < pitDelta && gap > 0;
  });
  // Update display
  posEl.textContent = 'P' + focused.classPosition;
  projEl.textContent = '→ P' + (focused.classPosition + wouldPass.length);
  // Show car names that would pass
});

Overlay structure:

- Two colored lines: focus driver (green) vs comparison driver (purple)

Data tracking:
```js
// Track lap times per driver as they complete laps
const driverLapHistory = {}; // carIdx -> [lapTime1, lapTime2, ...]
let compareMode = 'ahead'; // 'ahead' or 'behind'

onData('standings', data => {
  // For each driver, detect lap completion and record lap time
  data.forEach(d => {
    if (!driverLapHistory[d.carIdx]) driverLapHistory[d.carIdx] = { laps: [], lastLapCount: 0 };
    const hist = driverLapHistory[d.carIdx];
    if (d.lapsCompleted > hist.lastLapCount && d.lastLap > 0) {
      hist.laps.push(d.lastLap);
      hist.lastLapCount = d.lapsCompleted;
    }
  });

  // Find focus driver and comparison driver
  const focused = data.find(d => d.isSpectated) || data.find(d => d.isPlayer);
  if (!focused) return;
  const classmates = data.filter(d => d.carClass === focused.carClass)
    .sort((a, b) => a.classPosition - b.classPosition);
  const focusIdx = classmates.findIndex(d => d.carIdx === focused.carIdx);
  const compareIdx = compareMode === 'ahead' ? focusIdx - 1 : focusIdx + 1;
  const compareDriver = classmates[compareIdx];
  if (!compareDriver) return;

  // Build cumulative gap data and draw chart
  drawGapChart(focused.carIdx, compareDriver.carIdx, compareDriver.driverName);
});
```

Chart rendering: use HTML5 Canvas with:
- Dark grid background
- Green line for focus driver cumulative time
- Purple line for comparison driver
- Y-axis = gap in seconds (auto-scale)
- X-axis = lap number
